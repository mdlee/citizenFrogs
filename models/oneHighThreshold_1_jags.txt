# citizen frogs
# one high threshold, single frog
model{
  for (t in 1:nTrials){
    yHat[t, 1] = beta[person[t]]
    yHat[t, 2] = alpha[person[t]] + (1-alpha[person[t]])*beta[person[t]]
    y[t] ~ dbern(yHat[t, tau[stimulus[t]]+1])
  }
  for (i in 1:nStimuli){
    tau[i] ~ dbern(phi)
  }
  phi ~ dunif(0, 1)
  for (j in 1:nPeople){
    alpha[j] ~ dunif(0, 1)
    beta[j] ~ dunif(0, 1)
  }
}
